def recordAudio(FRAMES_PER_BUFFER=1024, FORMAT=pyaudio.paInt16, CHANNELS=1, RATE=44100, output_file_name="output.wav", seconds=5):
    p = pyaudio.PyAudio()       # creating pyAudio obect

    stream = p.open(format=FORMAT,              # öffnet den Datenstrom der Audio (wie wird die Audio abgetastet)
                    frames_per_buffer=FRAMES_PER_BUFFER,
                    channels=CHANNELS,
                    rate=RATE,
                    input=True)
    frames = []
    print("Press space to start recording.")
    keyboard.wait('space')
    print("recording...")

    for i in range(0, int(RATE / FRAMES_PER_BUFFER * seconds)):
        data = stream.read(FRAMES_PER_BUFFER)  # 1024 frames werden in einer Iteration gelesen
        frames.append(data)

    stream.stop_stream()
    stream.close()
    p.terminate()

    obj = wave.open(output_file_name, "wb")  # wb = write binary
    obj.setnchannels(CHANNELS)
    obj.setsampwidth(p.get_sample_size(FORMAT))
    obj.setframerate(RATE)
    obj.writeframes(b''.join(frames))
    obj.close()
